SELECT id_content, title_content, description_content, content.id_page,title_media,name_media
FROM event
INNER JOIN event_content
ON event_content.id_event=event.id_event
INNER JOIN content
ON content.id_content=event_content.id_content
INNER JOIN page
ON page.id_page=content.id_page
INNER JOIN media
ON page.id_page=media.id_page
ORDER BY end_date_event DESC LIMIT 1
